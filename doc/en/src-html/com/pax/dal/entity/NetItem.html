001package com.pax.dal.entity;
002
003/**
004 * <div class="zh"> 网络访问过滤规则类 </div> <div class="en"> Network access filtering rule class</div>
005 * @since V3.29.00
006 */
007public class NetItem {
008
009    private String addr;
010    private String dnsServer;
011    private int opType;
012
013
014    /**
015     * <div class="zh"> 创建一个NetItem的实例 </div> <div class="en"> Create an instance of NetItem</div>
016     */
017    public NetItem() {}
018
019    /**
020     * <div class="zh"> 创建一个NetItem的实例 </div> <div class="en"> Create an instance of NetItem</div>
021     * @param opType <div class="zh">类型：0代表IP，1代表域名 </div> <div class="en">Type: 0 for IP, 1 for domain name</div>
022     * @param addr <div class="zh">IP或域名地址 </div> <div class="en">IP or domain name address</div>
023     * @param dnsServer <div class="zh">Dns服务器地址，若opType为0，则该项无需关注 </div>
024     *                  <div class="en">DNS server address, if opType is 0, this item does not need to be concerned</div>
025     */
026    public NetItem(int opType, String addr, String dnsServer) {
027        this.opType = opType;
028        this.addr = addr;
029        this.dnsServer = dnsServer;
030    }
031
032    /**
033     * <div class="zh"> 获取域名或IP地址 </div> <div class="en"> Get domain name or IP address </div>
034     *
035     * @return addr
036     */
037    public String getAddr() {
038        return addr;
039    }
040
041    /**
042     * <div class="zh"> 设置域名或IP地址 </div> <div class="en"> Set domain name or IP address </div>
043     *
044     */
045    public void setAddr(String addr) {
046        this.addr = addr;
047    }
048
049    /**
050     * <div class="zh"> 获取DNS服务器 </div> <div class="en"> Get DNS server </div>
051     *
052     * @return dnsServer
053     */
054    public String getDnsServer() {
055        return dnsServer;
056    }
057
058    /**
059     * <div class="zh"> 设置DNS服务器 </div> <div class="en"> Set DNS server </div>
060     *
061     */
062    public void setDnsServer(String dnsServer) {
063        this.dnsServer = dnsServer;
064    }
065
066    /**
067     * <div class="zh"> 获取规则类型 </div> <div class="en"> Get rule type </div>
068     *
069     * @return opType
070     */
071    public int getOpType() {
072        return opType;
073    }
074
075    /**
076     * <div class="zh"> 设置规则类型 </div> <div class="en"> Set rule type </div>
077     *
078     */
079    public void setOpType(int opType) {
080        this.opType = opType;
081    }
082}
